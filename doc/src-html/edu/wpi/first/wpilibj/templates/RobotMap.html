007 * floating around.
008 */

011    public static final int RIGHT_MOTOR = 3;
012    public static final int LEFT_NET_MOTOR = 4;
013    public static final int RIGHT_NET_MOTOR = 5;
014    public static final int DUMP_MOTOR = 6;
015    public static final int RELEASE_MOTOR = 6;
016    
017

019}
